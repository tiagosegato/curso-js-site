Módulo 7: Noções de Banco de Dados

Fundamentos de armazenamento e manipulação de dados para aplicações backend

Visão Geral do Módulo

Neste módulo, você aprenderá os conceitos fundamentais de bancos de dados, incluindo modelagem, linguagem SQL, e como integrar bancos de dados relacionais e não-relacionais com aplicações Node.js. Esses conhecimentos são essenciais para desenvolver aplicações backend que necessitam armazenar e manipular dados de forma eficiente e segura.

Criação de Banco de Dados e Tabelas para Estúdio de Pilates

-- Criação do banco de dados
CREATE DATABASE IF NOT EXISTS estudio_pilates;

-- Uso do banco de dados
USE estudio_pilates;

-- Criação da tabela de alunos
CREATE TABLE IF NOT EXISTS alunos (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nome VARCHAR(100) NOT NULL,
  email VARCHAR(100) NOT NULL UNIQUE,
  telefone VARCHAR(20),
  data_nascimento DATE,
  endereco VARCHAR(200),
  observacoes TEXT,
  ativo BOOLEAN DEFAULT TRUE,
  data_cadastro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  data_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- Criação da tabela de instrutores
CREATE TABLE IF NOT EXISTS instrutores (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nome VARCHAR(100) NOT NULL,
  email VARCHAR(100) NOT NULL UNIQUE,
  telefone VARCHAR(20),
  especialidade VARCHAR(100),
  biografia TEXT,
  ativo BOOLEAN DEFAULT TRUE,
  data_cadastro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  data_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- Criação da tabela de modalidades
CREATE TABLE IF NOT EXISTS modalidades (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nome VARCHAR(100) NOT NULL,
  descricao TEXT,
  duracao_padrao INT COMMENT 'Duração padrão em minutos',
  ativo BOOLEAN DEFAULT TRUE,
  data_cadastro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  data_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- Criação da tabela de aulas
CREATE TABLE IF NOT EXISTS aulas (
  id INT AUTO_INCREMENT PRIMARY KEY,
  modalidade_id INT NOT NULL,
  instrutor_id INT NOT NULL,
  titulo VARCHAR(100) NOT NULL,
  descricao TEXT,
  data_hora_inicio DATETIME NOT NULL,
  data_hora_fim DATETIME NOT NULL,
  capacidade_maxima INT,
  ativo BOOLEAN DEFAULT TRUE,
  data_cadastro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  data_atualizacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (modalidade_id) REFERENCES modalidades(id),
  FOREIGN KEY (instrutor_id) REFERENCES instrutores(id)
);

-- Criação da tabela de matrículas (relacionamento entre alunos e aulas)
CREATE TABLE IF NOT EXISTS matriculas (
  id INT AUTO_INCREMENT PRIMARY KEY,
  aluno_id INT NOT NULL,
  aula_id INT NOT NULL,
  data_matricula TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  status ENUM('confirmada', 'cancelada', 'pendente') DEFAULT 'pendente',
  observacoes TEXT,
  FOREIGN KEY (aluno_id) REFERENCES alunos(id),
  FOREIGN KEY (aula_id) REFERENCES aulas(id),
  UNIQUE KEY unique_aluno_aula (aluno_id, aula_id)
);

-- Criação da tabela de pagamentos
CREATE TABLE IF NOT EXISTS pagamentos (
  id INT AUTO_INCREMENT PRIMARY KEY,
  aluno_id INT NOT NULL,
  valor DECIMAL(10, 2) NOT NULL,
  data_pagamento DATE NOT NULL,
  metodo_pagamento ENUM('dinheiro', 'cartao_credito', 'cartao_debito', 'pix', 'transferencia') NOT NULL,
  descricao VARCHAR(200),
  comprovante VARCHAR(255) COMMENT 'Caminho para o arquivo de comprovante',
  data_cadastro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (aluno_id) REFERENCES alunos(id)
);

-- Criação da tabela de avaliações físicas
CREATE TABLE IF NOT EXISTS avaliacoes (
  id INT AUTO_INCREMENT PRIMARY KEY,
  aluno_id INT NOT NULL,
  instrutor_id INT NOT NULL,
  data_avaliacao DATE NOT NULL,
  peso DECIMAL(5, 2),
  altura DECIMAL(3, 2),
  imc DECIMAL(5, 2),
  observacoes TEXT,
  recomendacoes TEXT,
  data_cadastro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (aluno_id) REFERENCES alunos(id),
  FOREIGN KEY (instrutor_id) REFERENCES instrutores(id)
);

Integração com Node.js e MySQL

// Importação dos módulos necessários
const mysql = require('mysql2/promise');
const dotenv = require('dotenv');

// Carregamento das variáveis de ambiente
dotenv.config();

// Configuração da conexão com o banco de dados
const dbConfig = {
  host: process.env.DB_HOST || 'localhost',
  user: process.env.DB_USER || 'root',
  password: process.env.DB_PASSWORD || 'password',
  database: process.env.DB_NAME || 'estudio_pilates',
  waitForConnections: true,
  connectionLimit: 10,
  queueLimit: 0
};

// Criação do pool de conexões
const pool = mysql.createPool(dbConfig);

// Classe para gerenciar alunos
class AlunoRepository {
  // Obter todos os alunos
  async findAll() {
    try {
      const [rows] = await pool.query('SELECT * FROM alunos WHERE ativo = TRUE ORDER BY nome');
      return rows;
    } catch (error) {
      console.error('Erro ao buscar alunos:', error);
      throw error;
    }
  }
  
  // Obter um aluno pelo ID
  async findById(id) {
    try {
      const [rows] = await pool.query('SELECT * FROM alunos WHERE id = ? AND ativo = TRUE', [id]);
      return rows[0] || null;
    } catch (error) {
      console.error(`Erro ao buscar aluno com ID ${id}:`, error);
      throw error;
    }
  }
  
  // Criar um novo aluno
  async create(aluno) {
    try {
      const { nome, email, telefone, data_nascimento, endereco, observacoes } = aluno;
      
      const [result] = await pool.query(
        'INSERT INTO alunos (nome, email, telefone, data_nascimento, endereco, observacoes) VALUES (?, ?, ?, ?, ?, ?)',
        [nome, email, telefone, data_nascimento, endereco, observacoes]
      );
      
      return { id: result.insertId, ...aluno };
    } catch (error) {
      console.error('Erro ao criar aluno:', error);
      throw error;
    }
  }
  
  // Atualizar um aluno existente
  async update(id, aluno) {
    try {
      const { nome, email, telefone, data_nascimento, endereco, observacoes } = aluno;
      
      await pool.query(
        'UPDATE alunos SET nome = ?, email = ?, telefone = ?, data_nascimento = ?, endereco = ?, observacoes = ? WHERE id = ?',
        [nome, email, telefone, data_nascimento, endereco, observacoes, id]
      );
      
      return this.findById(id);
    } catch (error) {
      console.error(`Erro ao atualizar aluno com ID ${id}:`, error);
      throw error;
    }
  }
  
  // Excluir um aluno (exclusão lógica)
  async delete(id) {
    try {
      await pool.query('UPDATE alunos SET ativo = FALSE WHERE id = ?', [id]);
      return { success: true, message: 'Aluno excluído com sucesso' };
    } catch (error) {
      console.error(`Erro ao excluir aluno com ID ${id}:`, error);
      throw error;
    }
  }
  
  // Buscar alunos por nome ou email
  async search(term) {
    try {
      const [rows] = await pool.query(
        'SELECT * FROM alunos WHERE ativo = TRUE AND (nome LIKE ? OR email LIKE ?)',
        [`%${term}%`, `%${term}%`]
      );
      return rows;
    } catch (error) {
      console.error(`Erro ao buscar alunos com termo "${term}":`, error);
      throw error;
    }
  }
  
  // Obter matrículas de um aluno
  async getMatriculas(alunoId) {
    try {
      const [rows] = await pool.query(`
        SELECT m.id, m.data_matricula, m.status, m.observacoes,
               a.id as aula_id, a.titulo, a.data_hora_inicio, a.data_hora_fim,
               i.nome as instrutor_nome, mo.nome as modalidade_nome
        FROM matriculas m
        JOIN aulas a ON m.aula_id = a.id
        JOIN instrutores i ON a.instrutor_id = i.id
        JOIN modalidades mo ON a.modalidade_id = mo.id
        WHERE m.aluno_id = ?
        ORDER BY a.data_hora_inicio DESC
      `, [alunoId]);
      
      return rows;
    } catch (error) {
      console.error(`Erro ao buscar matrículas do aluno com ID ${alunoId}:`, error);
      throw error;
    }
  }
}

// Exemplo de uso
async function exemploDeUso() {
  const alunoRepo = new AlunoRepository();
  
  try {
    // Listar todos os alunos
    console.log('Listando todos os alunos:');
    const alunos = await alunoRepo.findAll();
    console.log(alunos);
    
    // Criar um novo aluno
    console.log('\nCriando novo aluno:');
    const novoAluno = await alunoRepo.create({
      nome: 'Fernanda Lima',
      email: 'fernanda.lima@exemplo.com',
      telefone: '(11) 94444-3333',
      data_nascimento: '1988-09-25',
      endereco: 'Rua Bela Cintra, 500',
      observacoes: 'Iniciante em Pilates. Busca melhorar condicionamento físico.'
    });
    console.log('Novo aluno criado:', novoAluno);
    
    // Buscar aluno por ID
    console.log('\nBuscando aluno por ID:');
    const aluno = await alunoRepo.findById(novoAluno.id);
    console.log('Aluno encontrado:', aluno);
    
    // Atualizar aluno
    console.log('\nAtualizando aluno:');
    const alunoAtualizado = await alunoRepo.update(novoAluno.id, {
      ...aluno,
      telefone: '(11) 94444-5555',
      observacoes: 'Iniciante em Pilates. Tem experiência prévia com yoga.'
    });
    console.log('Aluno atualizado:', alunoAtualizado);
    
    // Buscar alunos por termo
    console.log('\nBuscando alunos por termo:');
    const alunosBuscados = await alunoRepo.search('Lima');
    console.log('Alunos encontrados:', alunosBuscados);
    
    // Excluir aluno
    console.log('\nExcluindo aluno:');
    const resultado = await alunoRepo.delete(novoAluno.id);
    console.log(resultado);
    
  } catch (error) {
    console.error('Erro no exemplo de uso:', error);
  } finally {
    // Encerrar o pool de conexões
    pool.end();
  }
}

// Executar o exemplo
exemploDeUso();
