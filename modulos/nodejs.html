Módulo 6: Node.js

Desenvolvimento de aplicações backend com JavaScript

Visão Geral do Módulo

Neste módulo, você aprenderá sobre Node.js, um ambiente de execução JavaScript do lado do servidor que permite construir aplicações backend escaláveis e de alta performance. Você conhecerá os fundamentos do Node.js, o framework Express para desenvolvimento de APIs e aplicações web, e como implementar funcionalidades essenciais para aplicações backend.

Exemplos Práticos

Servidor HTTP Básico com Node.js

// Importação do módulo http
const http = require('http');

// Definição da porta
const PORT = process.env.PORT || 3000;

// Criação do servidor
const server = http.createServer((req, res) => {
  // Configuração dos cabeçalhos de resposta
  res.setHeader('Content-Type', 'text/html');
  
  // Obtenção da URL da requisição
  const url = req.url;
  const method = req.method;
  
  // Roteamento básico
  if (url === '/') {
    res.statusCode = 200;
    res.end(`
      <html>
        <head>
          <title>Servidor Node.js</title>
        </head>
        <body>
          <h1>Bem-vindo ao Servidor Node.js</h1>
          <p>Esta é uma página inicial simples servida pelo Node.js.</p>
          <nav>
            <ul>
              <li><a href="/">Home</a></li>
              <li><a href="/sobre">Sobre</a></li>
              <li><a href="/contato">Contato</a></li>
            </ul>
          </nav>
        </body>
      </html>
    `);
  } else if (url === '/sobre') {
    res.statusCode = 200;
    res.end(`
      <html>
        <head>
          <title>Sobre - Servidor Node.js</title>
        </head>
        <body>
          <h1>Sobre o Servidor</h1>
          <p>Este é um exemplo simples de um servidor HTTP criado com Node.js.</p>
          <a href="/">Voltar para Home</a>
        </body>
      </html>
    `);
  } else if (url === '/contato') {
    res.statusCode = 200;
    res.end(`
      <html>
        <head>
          <title>Contato - Servidor Node.js</title>
        </head>
        <body>
          <h1>Contato</h1>
          <form method="POST" action="/enviar-mensagem">
            <div>
              <label for="nome">Nome:</label>
              <input type="text" id="nome" name="nome" required>
            </div>
            <div>
              <label for="email">Email:</label>
              <input type="email" id="email" name="email" required>
            </div>
            <div>
              <label for="mensagem">Mensagem:</label>
              <textarea id="mensagem" name="mensagem" required></textarea>
            </div>
            <button type="submit">Enviar</button>
          </form>
          <a href="/">Voltar para Home</a>
        </body>
      </html>
    `);
  } else if (url === '/enviar-mensagem' && method === 'POST') {
    // Captura dos dados do formulário
    let body = '';
    
    req.on('data', (chunk) => {
      body += chunk.toString();
    });
    
    req.on('end', () => {
      console.log('Dados do formulário:', body);
      
      // Redirecionamento após o envio
      res.statusCode = 302;
      res.setHeader('Location', '/mensagem-enviada');
      res.end();
    });
  } else if (url === '/mensagem-enviada') {
    res.statusCode = 200;
    res.end(`
      <html>
        <head>
          <title>Mensagem Enviada - Servidor Node.js</title>
        </head>
        <body>
          <h1>Mensagem Enviada com Sucesso!</h1>
          <p>Obrigado por entrar em contato. Responderemos em breve.</p>
          <a href="/">Voltar para Home</a>
        </body>
      </html>
    `);
  } else {
    // Página não encontrada
    res.statusCode = 404;
    res.end(`
      <html>
        <head>
          <title>404 - Página não encontrada</title>
        </head>
        <body>
          <h1>404 - Página não encontrada</h1>
          <p>A página que você está procurando não existe.</p>
          <a href="/">Voltar para Home</a>
        </body>
      </html>
    `);
  }
});

// Inicialização do servidor
server.listen(PORT, () => {
  console.log(`Servidor rodando em http://localhost:${PORT}`);
});

API RESTful com Express.js

// Importação dos módulos necessários
const express = require('express');
const bodyParser = require('body-parser');
const cors = require('cors');

// Criação da aplicação Express
const app = express();
const PORT = process.env.PORT || 3000;

// Configuração de middlewares
app.use(cors());
app.use(bodyParser.json());
app.use(bodyParser.urlencoded({ extended: true }));

// Middleware de logging
app.use((req, res, next) => {
  console.log(`${new Date().toISOString()} - ${req.method} ${req.url}`);
  next();
});

// Dados de exemplo (simulando um banco de dados)
let alunos = [
  { id: 1, nome: 'João Silva', email: 'joao@exemplo.com', telefone: '(11) 98765-4321', ativo: true },
  { id: 2, nome: 'Maria Santos', email: 'maria@exemplo.com', telefone: '(11) 91234-5678', ativo: true },
  { id: 3, nome: 'Pedro Oliveira', email: 'pedro@exemplo.com', telefone: '(11) 99876-5432', ativo: false }
];

// Rotas da API

// Rota raiz
app.get('/', (req, res) => {
  res.json({
    message: 'API de Gerenciamento de Alunos',
    version: '1.0.0',
    endpoints: [
      { method: 'GET', path: '/api/alunos', description: 'Listar todos os alunos' },
      { method: 'GET', path: '/api/alunos/:id', description: 'Obter um aluno específico' },
      { method: 'POST', path: '/api/alunos', description: 'Criar um novo aluno' },
      { method: 'PUT', path: '/api/alunos/:id', description: 'Atualizar um aluno existente' },
      { method: 'DELETE', path: '/api/alunos/:id', description: 'Excluir um aluno' }
    ]
  });
});

// Listar todos os alunos
app.get('/api/alunos', (req, res) => {
  // Filtragem opcional por status (ativo/inativo)
  const { ativo } = req.query;
  
  if (ativo !== undefined) {
    const filtrado = alunos.filter(aluno => aluno.ativo === (ativo === 'true'));
    return res.json({
      success: true,
      count: filtrado.length,
      data: filtrado
    });
  }
  
  res.json({
    success: true,
    count: alunos.length,
    data: alunos
  });
});

// Obter um aluno específico
app.get('/api/alunos/:id', (req, res) => {
  const id = parseInt(req.params.id);
  const aluno = alunos.find(a => a.id === id);
  
  if (!aluno) {
    return res.status(404).json({
      success: false,
      message: 'Aluno não encontrado'
    });
  }
  
  res.json({
    success: true,
    data: aluno
  });
});

// Criar um novo aluno
app.post('/api/alunos', (req, res) => {
  const { nome, email, telefone } = req.body;
  
  // Validação básica
  if (!nome || !email) {
    return res.status(400).json({
      success: false,
      message: 'Nome e email são obrigatórios'
    });
  }
  
  // Verificar se o email já existe
  if (alunos.some(a => a.email === email)) {
    return res.status(409).json({
      success: false,
      message: 'Email já cadastrado'
    });
  }
  
  // Criar novo aluno
  const novoAluno = {
    id: alunos.length > 0 ? Math.max(...alunos.map(a => a.id)) + 1 : 1,
    nome,
    email,
    telefone,
    ativo: true
  };
  
  alunos.push(novoAluno);
  
  res.status(201).json({
    success: true,
    message: 'Aluno criado com sucesso',
    data: novoAluno
  });
});

// Atualizar um aluno existente
app.put('/api/alunos/:id', (req, res) => {
  const id = parseInt(req.params.id);
  const { nome, email, telefone, ativo } = req.body;
  
  // Encontrar o índice do aluno
  const index = alunos.findIndex(a => a.id === id);
  
  if (index === -1) {
    return res.status(404).json({
      success: false,
      message: 'Aluno não encontrado'
    });
  }
  
  // Verificar se o email já existe (exceto para o próprio aluno)
  if (email && email !== alunos[index].email && alunos.some(a => a.email === email)) {
    return res.status(409).json({
      success: false,
      message: 'Email já cadastrado para outro aluno'
    });
  }
  
  // Atualizar o aluno
  alunos[index] = {
    ...alunos[index],
    nome: nome || alunos[index].nome,
    email: email || alunos[index].email,
    telefone: telefone || alunos[index].telefone,
    ativo: ativo !== undefined ? ativo : alunos[index].ativo
  };
  
  res.json({
    success: true,
    message: 'Aluno atualizado com sucesso',
    data: alunos[index]
  });
});

// Excluir um aluno
app.delete('/api/alunos/:id', (req, res) => {
  const id = parseInt(req.params.id);
  
  // Verificar se o aluno existe
  if (!alunos.some(a => a.id === id)) {
    return res.status(404).json({
      success: false,
      message: 'Aluno não encontrado'
    });
  }
  
  // Remover o aluno
  alunos = alunos.filter(a => a.id !== id);
  
  res.json({
    success: true,
    message: 'Aluno excluído com sucesso'
  });
});

// Middleware para tratamento de rotas não encontradas
app.use((req, res) => {
  res.status(404).json({
    success: false,
    message: 'Endpoint não encontrado'
  });
});

// Middleware para tratamento de erros
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).json({
    success: false,
    message: 'Erro interno do servidor'
  });
});

// Inicialização do servidor
app.listen(PORT, () => {
  console.log(`Servidor rodando em http://localhost:${PORT}`);
});

Exercícios Práticos

Para fixar os conceitos aprendidos neste módulo, recomendamos a realização dos seguintes exercícios:

  1. Crie um servidor HTTP básico que sirva páginas HTML estáticas.
  2. Desenvolva uma API RESTful para gerenciar uma lista de tarefas (to-do list).
  3. Implemente autenticação JWT em uma API Express.
  4. Crie um middleware personalizado para logging de requisições.
  5. Desenvolva uma aplicação que leia e escreva arquivos utilizando o módulo fs.
  6. Implemente uma API que se conecte a um banco de dados (MySQL ou MongoDB).
  7. Crie testes unitários para uma API Express utilizando Jest ou Mocha.
  8. Desenvolva uma aplicação que utilize WebSockets para comunicação em tempo real.

Você encontrará mais exercícios e desafios na seção de Exercícios.
