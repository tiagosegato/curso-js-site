Módulo 5: Introdução a WebServices

Fundamentos de comunicação entre sistemas através da web

Visão Geral do Módulo

Neste módulo, você aprenderá sobre WebServices, que são sistemas de software projetados para suportar a interoperabilidade entre diferentes aplicações através da internet. Você conhecerá os principais tipos de WebServices, protocolos de comunicação e formatos de dados utilizados para a troca de informações entre sistemas distribuídos.

Exemplos Práticos

Criação de uma API RESTful com Express

// Importação dos módulos necessários
const express = require('express');
const bodyParser = require('body-parser');

// Criação da aplicação Express
const app = express();
const port = 3000;

// Middleware para parsing de JSON
app.use(bodyParser.json());

// Dados de exemplo (simulando um banco de dados)
let alunos = [
  { id: 1, nome: 'João Silva', email: 'joao@exemplo.com', telefone: '(11) 98765-4321', ativo: true },
  { id: 2, nome: 'Maria Santos', email: 'maria@exemplo.com', telefone: '(11) 91234-5678', ativo: true },
  { id: 3, nome: 'Pedro Oliveira', email: 'pedro@exemplo.com', telefone: '(11) 99876-5432', ativo: false }
];

// Rota para obter todos os alunos
app.get('/api/alunos', (req, res) => {
  // Filtragem opcional por status (ativo/inativo)
  const { ativo } = req.query;
  
  if (ativo !== undefined) {
    const filtrado = alunos.filter(aluno => aluno.ativo === (ativo === 'true'));
    return res.json({
      success: true,
      message: 'Alunos recuperados com sucesso',
      data: filtrado
    });
  }
  
  res.json({
    success: true,
    message: 'Alunos recuperados com sucesso',
    data: alunos
  });
});

// Rota para obter um aluno específico por ID
app.get('/api/alunos/:id', (req, res) => {
  const id = parseInt(req.params.id);
  const aluno = alunos.find(a => a.id === id);
  
  if (!aluno) {
    return res.status(404).json({
      success: false,
      message: 'Aluno não encontrado'
    });
  }
  
  res.json({
    success: true,
    message: 'Aluno recuperado com sucesso',
    data: aluno
  });
});

// Rota para criar um novo aluno
app.post('/api/alunos', (req, res) => {
  const { nome, email, telefone } = req.body;
  
  // Validação básica
  if (!nome || !email) {
    return res.status(400).json({
      success: false,
      message: 'Nome e email são obrigatórios'
    });
  }
  
  // Verificar se o email já existe
  if (alunos.some(a => a.email === email)) {
    return res.status(409).json({
      success: false,
      message: 'Email já cadastrado'
    });
  }
  
  // Criar novo aluno
  const novoAluno = {
    id: alunos.length > 0 ? Math.max(...alunos.map(a => a.id)) + 1 : 1,
    nome,
    email,
    telefone,
    ativo: true
  };
  
  alunos.push(novoAluno);
  
  res.status(201).json({
    success: true,
    message: 'Aluno criado com sucesso',
    data: novoAluno
  });
});

// Rota para atualizar um aluno existente
app.put('/api/alunos/:id', (req, res) => {
  const id = parseInt(req.params.id);
  const { nome, email, telefone, ativo } = req.body;
  
  // Encontrar o índice do aluno
  const index = alunos.findIndex(a => a.id === id);
  
  if (index === -1) {
    return res.status(404).json({
      success: false,
      message: 'Aluno não encontrado'
    });
  }
  
  // Verificar se o email já existe (exceto para o próprio aluno)
  if (email && email !== alunos[index].email && alunos.some(a => a.email === email)) {
    return res.status(409).json({
      success: false,
      message: 'Email já cadastrado para outro aluno'
    });
  }
  
  // Atualizar o aluno
  alunos[index] = {
    ...alunos[index],
    nome: nome || alunos[index].nome,
    email: email || alunos[index].email,
    telefone: telefone || alunos[index].telefone,
    ativo: ativo !== undefined ? ativo : alunos[index].ativo
  };
  
  res.json({
    success: true,
    message: 'Aluno atualizado com sucesso',
    data: alunos[index]
  });
});

// Rota para excluir um aluno
app.delete('/api/alunos/:id', (req, res) => {
  const id = parseInt(req.params.id);
  
  // Verificar se o aluno existe
  if (!alunos.some(a => a.id === id)) {
    return res.status(404).json({
      success: false,
      message: 'Aluno não encontrado'
    });
  }
  
  // Remover o aluno
  alunos = alunos.filter(a => a.id !== id);
  
  res.json({
    success: true,
    message: 'Aluno excluído com sucesso'
  });
});

// Iniciar o servidor
app.listen(port, () => {
  console.log(`Servidor rodando em http://localhost:${port}`);
});

Consumo de API RESTful com Fetch API

// URL base da API
const API_URL = 'http://localhost:3000/api';

// Função para obter todos os alunos
async function obterAlunos() {
  try {
    const response = await fetch(`${API_URL}/alunos`);
    
    if (!response.ok) {
      throw new Error(`Erro HTTP: ${response.status}`);
    }
    
    const data = await response.json();
    return data.data; // Retorna apenas o array de alunos
  } catch (error) {
    console.error('Erro ao obter alunos:', error);
    throw error;
  }
}

// Função para obter um aluno específico
async function obterAlunoPorId(id) {
  try {
    const response = await fetch(`${API_URL}/alunos/${id}`);
    
    if (response.status === 404) {
      return null; // Aluno não encontrado
    }
    
    if (!response.ok) {
      throw new Error(`Erro HTTP: ${response.status}`);
    }
    
    const data = await response.json();
    return data.data; // Retorna apenas o objeto do aluno
  } catch (error) {
    console.error(`Erro ao obter aluno ${id}:`, error);
    throw error;
  }
}

// Função para criar um novo aluno
async function criarAluno(aluno) {
  try {
    const response = await fetch(`${API_URL}/alunos`, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(aluno)
    });
    
    if (!response.ok) {
      const errorData = await response.json();
      throw new Error(errorData.message || `Erro HTTP: ${response.status}`);
    }
    
    const data = await response.json();
    return data.data; // Retorna o aluno criado
  } catch (error) {
    console.error('Erro ao criar aluno:', error);
    throw error;
  }
}

// Função para atualizar um aluno existente
async function atualizarAluno(id, dadosAtualizados) {
  try {
    const response = await fetch(`${API_URL}/alunos/${id}`, {
      method: 'PUT',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(dadosAtualizados)
    });
    
    if (response.status === 404) {
      throw new Error('Aluno não encontrado');
    }
    
    if (!response.ok) {
      const errorData = await response.json();
      throw new Error(errorData.message || `Erro HTTP: ${response.status}`);
    }
    
    const data = await response.json();
    return data.data; // Retorna o aluno atualizado
  } catch (error) {
    console.error(`Erro ao atualizar aluno ${id}:`, error);
    throw error;
  }
}

// Função para excluir um aluno
async function excluirAluno(id) {
  try {
    const response = await fetch(`${API_URL}/alunos/${id}`, {
      method: 'DELETE'
    });
    
    if (response.status === 404) {
      throw new Error('Aluno não encontrado');
    }
    
    if (!response.ok) {
      const errorData = await response.json();
      throw new Error(errorData.message || `Erro HTTP: ${response.status}`);
    }
    
    return true; // Exclusão bem-sucedida
  } catch (error) {
    console.error(`Erro ao excluir aluno ${id}:`, error);
    throw error;
  }
}

// Exemplo de uso das funções
async function exemploDeUso() {
  try {
    // Obter todos os alunos
    console.log('Obtendo todos os alunos...');
    const alunos = await obterAlunos();
    console.log('Alunos:', alunos);
    
    // Criar um novo aluno
    console.log('\nCriando novo aluno...');
    const novoAluno = await criarAluno({
      nome: 'Ana Souza',
      email: 'ana@exemplo.com',
      telefone: '(11) 97777-8888'
    });
    console.log('Novo aluno criado:', novoAluno);
    
    // Obter um aluno específico
    console.log('\nObtendo aluno específico...');
    const aluno = await obterAlunoPorId(novoAluno.id);
    console.log('Aluno encontrado:', aluno);
    
    // Atualizar um aluno
    console.log('\nAtualizando aluno...');
    const alunoAtualizado = await atualizarAluno(novoAluno.id, {
      telefone: '(11) 96666-7777'
    });
    console.log('Aluno atualizado:', alunoAtualizado);
    
    // Excluir um aluno
    console.log('\nExcluindo aluno...');
    await excluirAluno(novoAluno.id);
    console.log('Aluno excluído com sucesso');
    
    // Verificar se o aluno foi excluído
    console.log('\nVerificando exclusão...');
    const alunoExcluido = await obterAlunoPorId(novoAluno.id);
    console.log('Aluno após exclusão:', alunoExcluido); // Deve ser null
    
  } catch (error) {
    console.error('Erro no exemplo de uso:', error);
  }
}

// Executar o exemplo
exemploDeUso();

Exercícios Práticos

Para fixar os conceitos aprendidos neste módulo, recomendamos a realização dos seguintes exercícios:

  1. Crie uma API RESTful simples com Express para gerenciar uma lista de tarefas (to-do list).
  2. Implemente autenticação JWT em uma API existente.
  3. Consuma uma API pública (como OpenWeatherMap, GitHub API, etc.) utilizando Fetch API ou Axios.
  4. Crie uma documentação para uma API utilizando Swagger/OpenAPI.
  5. Implemente paginação, filtragem e ordenação em uma API RESTful.
  6. Crie um cliente para consumir um serviço SOAP existente.
  7. Implemente uma API GraphQL simples para um catálogo de produtos.
  8. Desenvolva um middleware de autenticação para uma API Express.

Você encontrará mais exercícios e desafios na seção de Exercícios.
