Módulo 4: Padrões de Desenvolvimento de Software

Soluções elegantes para problemas recorrentes no desenvolvimento de software

Visão Geral do Módulo

Exemplos Práticos

Padrão Singleton

// Implementação do padrão Singleton em JavaScript
class DatabaseConnection {
  constructor() {
    if (DatabaseConnection.instance) {
      return DatabaseConnection.instance;
    }
    
    // Simulação de configuração de conexão com banco de dados
    this.host = 'localhost';
    this.username = 'admin';
    this.password = 'password';
    this.database = 'myapp';
    this.connection = null;
    
    // Armazena a instância
    DatabaseConnection.instance = this;
  }
  
  connect() {
    if (this.connection) {
      console.log('Já conectado ao banco de dados');
      return this.connection;
    }
    
    console.log(`Conectando ao banco de dados ${this.database} em ${this.host}...`);
    // Simulação de conexão
    this.connection = {
      id: Math.floor(Math.random() * 1000),
      timestamp: new Date().toISOString()
    };
    
    console.log(`Conexão estabelecida com ID: ${this.connection.id}`);
    return this.connection;
  }
  
  disconnect() {
    if (this.connection) {
      console.log(`Desconectando do banco de dados (ID: ${this.connection.id})...`);
      this.connection = null;
      console.log('Desconectado com sucesso');
    } else {
      console.log('Não há conexão ativa para desconectar');
    }
  }
  
  executeQuery(query) {
    if (!this.connection) {
      console.log('Não há conexão ativa. Conectando primeiro...');
      this.connect();
    }
    
    console.log(`Executando query: ${query}`);
    return `Resultado da query: ${query}`;
  }
}

// Uso do Singleton
function testSingleton() {
  console.log('Criando primeira instância...');
  const db1 = new DatabaseConnection();
  db1.connect();
  
  console.log('\nCriando segunda instância...');
  const db2 = new DatabaseConnection();
  
  console.log('\nVerificando se as instâncias são as mesmas:');
  console.log(`db1 === db2: ${db1 === db2}`);
  
  console.log('\nExecutando query na primeira instância:');
  db1.executeQuery('SELECT * FROM users');
  
  console.log('\nExecutando query na segunda instância:');
  db2.executeQuery('SELECT * FROM products');
  
  console.log('\nDesconectando usando a segunda instância:');
  db2.disconnect();
  
  console.log('\nTentando executar query após desconexão:');
  db1.executeQuery('SELECT * FROM orders');
}

testSingleton();

Padrão Observer

// Implementação do padrão Observer em JavaScript
class Subject {
  constructor() {
    this.observers = [];
  }
  
  subscribe(observer) {
    if (!this.observers.includes(observer)) {
      this.observers.push(observer);
      console.log('Observer adicionado');
    }
  }
  
  unsubscribe(observer) {
    const index = this.observers.indexOf(observer);
    if (index !== -1) {
      this.observers.splice(index, 1);
      console.log('Observer removido');
    }
  }
  
  notify(data) {
    console.log('Notificando observers...');
    this.observers.forEach(observer => observer.update(data));
  }
}

class Observer {
  constructor(name) {
    this.name = name;
  }
  
  update(data) {
    console.log(`${this.name} recebeu atualização: ${JSON.stringify(data)}`);
  }
}

// Exemplo de uso: Sistema de notificações
class NotificationSystem extends Subject {
  constructor() {
    super();
    this.notifications = [];
  }
  
  addNotification(notification) {
    this.notifications.push({
      id: this.notifications.length + 1,
      ...notification,
      timestamp: new Date().toISOString()
    });
    
    console.log(`Nova notificação: ${notification.message}`);
    this.notify(notification);
  }
  
  getNotifications() {
    return this.notifications;
  }
}

// Uso do padrão Observer
function testObserver() {
  // Criação do sistema de notificações
  const notificationSystem = new NotificationSystem();
  
  // Criação de observers
  const userObserver = new Observer('Usuário');
  const adminObserver = new Observer('Administrador');
  const logObserver = new Observer('Sistema de Log');
  
  // Inscrição dos observers
  notificationSystem.subscribe(userObserver);
  notificationSystem.subscribe(adminObserver);
  notificationSystem.subscribe(logObserver);
  
  // Adição de notificações
  notificationSystem.addNotification({
    type: 'info',
    message: 'Sistema atualizado para a versão 2.0'
  });
  
  // Cancelamento da inscrição do usuário
  console.log('\nRemovendo usuário das notificações...\n');
  notificationSystem.unsubscribe(userObserver);
  
  // Nova notificação (não será recebida pelo usuário)
  notificationSystem.addNotification({
    type: 'alert',
    message: 'Manutenção programada para amanhã às 22h'
  });
  
  // Listagem de todas as notificações
  console.log('\nTodas as notificações:');
  console.log(notificationSystem.getNotifications());
}

testObserver();

Exercícios Práticos

Para fixar os conceitos aprendidos neste módulo, recomendamos a realização dos seguintes exercícios:

  1. Implemente o padrão Singleton para gerenciar a conexão com um banco de dados.

  3. Desenvolva uma fábrica de objetos utilizando o padrão Factory Method.

  5. Crie uma API simples utilizando o padrão MVC.

  7. Implemente o padrão Repository para acesso a dados em uma aplicação Node.js.
  8. Refatore um código existente aplicando os princípios SOLID.
